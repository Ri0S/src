버팀목                                                                              白雲/서문인                                                                       봉우리를 품고                                                                      산맥을 더듬고 지나서                                                                      깊은 골짜기를 흔들어 놓고                                                                      벌판을 내 달리는                                                                      태풍이고 싶다                                                                       그대의 아픔을 씻어 주고                                                                      억센 다짐을 꿀꺽 삼키며                                                                      묵직하게 떨어지는                                                                       눈물이고 싶다                                                                       홀로서는 이별을 걱정하며                                                                      힘들게 맞추어 살아가는                                                                       너의 손을 잡아주고                                                                      이야기를 함께 써가는                                                                       연인이고 싶다                                                                       우연처럼 편하고                                                                      미련처럼 남아                                                                      글썽거리는                                                                       진정한 사랑이고 싶다
