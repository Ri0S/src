가을 그리움 / 최홍윤                                         가을,                                        눈부신 날에는 언제나                                         쪽빛 하늘에                                         조개구름 하나 띄워놓고                                        습관처럼                                         너를 그리워했다.                                         코스모스 길섶에                                        소슬바람처럼 날아든                                        상큼한 네 소식에                                        난, 그만                                         눈이 시리도록 쳐다본                                        파란 하늘에 미안했다                                         늘 그리움이                                        내 가슴에 사무칠 때면                                        조각구름은 네에게로 가고                                          무작정 널 그리워만 하던                                         내가                                        강한 건지                                         약한 건지 모르나                                        지금은                                         참                                         그리울 때다.
