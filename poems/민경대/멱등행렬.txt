C = A x B.   행렬의 제곱을 계산해주는 함수pow(DoubleMatrix2D A, int p)   : Linear algebraic matrix power; B = Ak <==> B = A*A*...*A.   위의 함수 중 하나를 선택하여 사용하여  A²를 계산하여 A² = A임을 확인하면 될 것이다.       [소스코드]   [matrixTest.java]  :소스코드 파일 첨부 import cern.colt.matrix.impl.DenseDoubleMatrix2D;import cern.colt.matrix.linalg.Algebra;   public class matrixTest {     public static void main(String args[]){              //증명할 행렬 A : DenseDoubleMatrix2D(double[][] values)           DenseDoubleMatrix2D a = new DenseDoubleMatrix2D(new double[][]                                                                                  {{2,-3,-5},{-1,4,5},{1,-3,-4}});            //계산한 값을 저장할 행렬 B :DenseDoubleMatrix2D(int rows, int columns)           DenseDoubleMatrix2D b = new DenseDoubleMatrix2D(3,3);              Algebra alg = new Algebra(); //Algebra 객체 생성           //행렬 A 출력          System.out.println("A: "+a+"\n");              //행렬의 제곱 : pow(DoubleMatrix2D A, int p)           b.assign(alg.pow(a,2));  //A²을 계산하여 b에 저장          System.out.println("A²: "+b+"\n");             /**          * 멱등행렬인지 여부 출력          */          if(a.equals(b)==true){  //행렬 A와  B가 같으면                System.out.println("결과: A는 멱등행렬입니다.");          } else {  //행렬 A와  B가 같지 않으면                System.out.println("결과: A는 멱등행렬이 아닙니다.");          }       }}       [실행결과]           [결론]   A는 멱등행렬이다. [출처] matrix - 멱등행렬 증명 (두춘호)|작성자 유노추노  http://duru1024.blog.me/70016747977
