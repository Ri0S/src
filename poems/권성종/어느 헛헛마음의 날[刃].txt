어느 헛헛마음의 날[刃]                                          동서남북 돌아 보아도 헛보이는                                        하늘 덮은 헛안개                                        안개가 그린 헛원을 가위로 오려                                         너도나도 헛도는 마음밭에                                        물레방아 돌리거늘                                        어찌 샘물솟듯 헛물음 하는가                                         매 밤 자고싶은 어느 내 헛밤                                        허공을 질러가도 헛 가며                                         허심을 날아가도 헛 날아가는                                         뭔지 몰라도 쓰려 아픈                                         헛 마음 자라 헛헛한 빈 뇌고惱苦                                        그게 참 헛이라면 오히려 편하련만                                         꿈에도 없는 헛 아닌가                                         시때없이 불연듯 튀쳐 헛상면相面                                          공기같은 헛소리                                         뭘 맞닿아도 헛 찿아 어쩌랴                                          내 눈알 헛피를 내뿜으랴                                        윗층 또 그 윗층부터                                           허물어 부서져 내리는 헛건물                                        뒤틀어진 내림 계단의 헛바닥                                        다시 헛바닥 밑에 꾸겨그린                                        비인 차원의 헛땀 그림                                        뭐 있으며 어찌 있는                                         내 가슴의 헛골가루                                        헛헛하기만한 가루 펄펄 날리는                                         이 의미는 뭔가                                         헛뇌고에 목메는 헛헛말의 날[刃]                                        나도 당신도 모두 헛 알아온                 죽음보다 써늘한 헛헛깨비의 진이다
