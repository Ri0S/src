                                해너머 낡은 공기 여닫는 딸꾹질                                                            밥상 위 밥풀 갉아먹고 있을 때,                                                   눈빛 서로 엮어 뽑아내는 오늘 아바타                                                     옆 상에 번져 커지는 환호소리 곁 곁,                                          눈             내 귀 언저리 불거지는 스트레스                                           눈 깜박일 때마다 날카로운 경고음이 들린다                                                                                         꿀쩝꿀쩝 돼지들 처먹는 소린가                                                                                         피둥피둥 숫돼지 배 튕기는 소리인가 봐                                                                                          쩝쩝                                                                                  아직 사랑소리 담고 있을 공간이 남아있는가                                                              패대기쳐 멱살 쥐어 짜내어도                                      시장터 미각과 기억의 무늬는 산화되어 빛 바랜                                                                          쓰레기더미 들추는 짓이거늘                                                       상점, 금괴, 웃음, 활자가 그리움으로 그늘지면                                                          이미 멸실로 통하는 길 목                                                           거칠은 무사가 신작로 뚫는 탐험 길의 미스터리,                                                                 저멀리 망원 렌즈 뻗어, 쏘는                                                                                         어린 새끼 들에게 먹이는 한 방 주먹 밥이며                                                                               엇비슷 범선에서 내리도록, 배 밑창을 뚫는 탄알이다                                                                   쥐락펴락 신작로 닦아,                                                  짧고 뾰족한 칼꽃을 그려내는 필법, 화면을 자른다                                                             새 턴널을 뚫는 드릴 불꽃,                                                                                          침이 강철 컨트롤 박스를 꿰뚫으며                                                                                         남은 정자로 사선을 뚫어 홰치며                                                                                         불임에도 남의 콧물 훔쳐 수정난을 뚫는다                                              나이테 길거나 짧거나 배주리면 거지,                                          맞짱뜰 수록 상쾌하게 바람부는 옛 뱃사람들 뱃길이니                                                                               스스로 촛농 짜, 새끼 들이 신작로를 질주하게                                                                            발기하는 생식, 빗금으로 표시한 새빛 무늬 그리다
