.                                          웃음을 뚫으면 그리움이다                     해 저물녘 하늘에 시간빛이 차다                                          오랜 시간 차게 식어                                            빛 그 줄기에 덕지덕지 옛정의 때가 낀                                           낡은 얼굴            너의 오랜 망부석 하나 언 바람 소리 가른다                                                                    그 그림자를 봐라                                           오                                          우리 사이 넘치던 정이 얼어붙은 꿈이다                                                            그래                                           우리 그림자                꿈 속에 너의 눈물이 드나들더니 녹았다                                          한 점 유성으로 촉을 빚어                                          하늘 끝에서                                          내 닫힌 창문에 빛으로 낙하한다                          손에 장미를 그려주며                            목에 별을 조각하고                  핏줄에 맑은 관을 심어주며                                          그림자                                           뾰족하게 갈아 새하얀 소망                                          날아                                          멀리 아득하던 내 웃음을 꿰뚫는다                 명중되었습니다                       실은                          아             언제부터 내가 태우는 그리움입니다                                           오르는 곳도 하늘이 아니며                                            날개 저어가는 길도 깊은 그리움입니다
